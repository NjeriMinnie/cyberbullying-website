        function buildDeck() {

            currentDeck = [];

            let cardId = 1;

           

            for (const [category, cards] of Object.entries(cardData)) {

                if (currentFilter === 'all' || currentFilter === category) {

                    cards.forEach((card, index) => {

                        currentDeck.push({

                            id: cardId++,

                            category: category,

                            number: index + 1,

                            ...card

                        });

                    });

                }

            }

        }

 

        function shuffleDeck() {

            // Fisher-Yates shuffle

            for (let i = currentDeck.length - 1; i > 0; i--) {

                const j = Math.floor(Math.random() * (i + 1));

                [currentDeck[i], currentDeck[j]] = [currentDeck[j], currentDeck[i]];

            }

            updateUI();

        }

 

        function drawCard() {

            if (currentDeck.length === 0) {

                alert('No more cards in the deck! Click "Reset Deck" to start over.');

                return;

            }

 

            const card = currentDeck.pop();

            drawnCards.push(card);

           

            // Add to history

            cardHistory.unshift(card);

            if (cardHistory.length > 5) {

                cardHistory.pop();

            }

 

            displayCard(card);

            updateUI();

        }

            cardElement.querySelector('.card-category').className = `card-category ${card.category}`;

            cardElement.querySelector('.card-number').textContent = `Card ${card.number}`;

            cardElement.querySelector('.card-main-text').textContent = card.text;

           

            const childQuestion = cardElement.querySelector('.child-asks .question-text');

            const parentQuestion = cardElement.querySelector('.parent-asks .question-text');

            childQuestion.textContent = card.childAsk;

            parentQuestion.textContent = card.parentAsk;

           

            // Show card

            cardElement.classList.remove('hidden');

            cardElement.classList.add('showing');

        }

 

        function resetDeck() {

            drawnCards = [];

            cardHistory = [];

            buildDeck();

            shuffleDeck();

           

            // Hide current card and show placeholder

            const cardElement = document.getElementById('currentCard');

            const placeholder = document.getElementById('deckPlaceholder');

           

            cardElement.classList.remove('showing');

            cardElement.classList.add('hidden');

            placeholder.style.display = 'block';

           

            updateUI();

        }

 

        function setFilter(category) {

            currentFilter = category;

           

            // Update filter buttons

            document.querySelectorAll('.filter-btn').forEach(btn => {

                btn.classList.toggle('active', btn.dataset.category === category);

            });

           

            resetDeck();

        }

 

        function updateUI() {

            // Update cards remaining

            document.getElementById('cardsRemaining').textContent = currentDeck.length;

           

            // Update card history

            const historyContainer = document.getElementById('cardHistory');

            if (cardHistory.length === 0) {

                historyContainer.innerHTML = '<p class="no-history">No cards drawn yet. Start the conversation!</p>';

            } else {

                historyContainer.innerHTML = cardHistory.map(card => `

                    <div class="history-item">

                        <span class="history-category">${card.category}</span>

                        <div style="font-weight: bold; margin: 0.5rem 0;">${card.text}</div>

                    </div>

                `).join('');

            }

        }

 

        function setupEventListeners() {

            // Game controls

            const drawBtn = document.getElementById('drawCard');

            const shuffleBtn = document.getElementById('shuffleDeck');

            const resetBtn = document.getElementById('resetDeck');

           

            if (drawBtn) drawBtn.addEventListener('click', drawCard);

            if (shuffleBtn) shuffleBtn.addEventListener('click', shuffleDeck);

            if (resetBtn) resetBtn.addEventListener('click', resetDeck);

           

            // Filter buttons

            document.querySelectorAll('.filter-btn').forEach(btn => {

                btn.addEventListener('click', (e) => {

                    setFilter(e.target.dataset.category);

                });

            });

        }

 

        function toggleMobileMenu() {

            const navMenu = document.getElementById('navMenu');

            if (navMenu) {

                navMenu.classList.toggle('active');

            }

        }

 

        // Close mobile menu when clicking outside

        document.addEventListener('click', function(event) {

            const navMenu = document.getElementById('navMenu');

            const toggle = document.querySelector('.mobile-menu-toggle');

           

            if (navMenu && toggle && !navMenu.contains(event.target) && !toggle.contains(event.target)) {

                navMenu.classList.remove('active');

            }

        });
